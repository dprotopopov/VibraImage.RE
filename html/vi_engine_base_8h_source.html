44 inline T Median(std::list<T>& l,size_t limit)
45 {
46  while(l.size() > limit)
47  l.pop_front();
48  if(l.empty())
49  return 0;

51  std::list<T> tmp = l;
52  tmp.sort();
53  while(tmp.size() > 2)

55  tmp.pop_back();

57 }
